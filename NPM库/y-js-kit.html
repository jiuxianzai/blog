功能模块 ​

isKit 判断处理 ​

  • 是否为空
  • 是否不为空
  • 是否为数组
  • 是否为字符串
  • 是否为合法邮箱
  • 是否为合法地址
  • 是否为合法中国大陆身份证
  • 是否相等
js
import Lib from "y-js-kit";

console.log('============ isKit ============');
console.log('-- isEmpty 是否空 --');
console.log(Lib.is.isEmpty(null)); // true
console.log(Lib.is.isEmpty("233")); // false

console.log('-- isNotEmpty 是否不为空 --');
console.log(Lib.is.isNotEmpty(undefined)); // false
console.log(Lib.is.isNotEmpty("2333")); // true

console.log('-- isArray 是否数组 --');
console.log(Lib.is.isArray([1, 2, 3, 4])); // true
console.log(Lib.is.isArray("123")); // false

console.log('-- isString 是否字符串 --');
console.log(Lib.is.isString([1, 2, 3, 4])); // false
console.log(Lib.is.isString("123")); // true

console.log('-- isEmail 是否合法邮箱 --');
console.log(Lib.is.isEmail("youyanping@example.com")); // true
console.log(Lib.is.isEmail("123")); // false

console.log('-- isURL 是否合法地址 --');
console.log(Lib.is.isURL("https://www.baidu.com")); // true
console.log(Lib.is.isURL("123")); // false

console.log('-- isIdCard 是否合法中国大陆身份证 --');
console.log(Lib.is.isIdCard("")); // true
console.log(Lib.is.isIdCard("123")); // false

console.log('-- isEqual 是否相等 --');
console.log(Lib.is.isEqual("12", "12")); // true
console.log(Lib.is.isEqual({age: 20, name: "zs"}, {age: 21, name: "zs"})); // false

urlKit 参数处理 ​

  • 获取地址栏全部参数
  • 获取地址栏指定参数
  • 添加地址栏参数
  • 转换地址栏参数
js
import Lib from "y-js-kit";

console.log('============ urlKit ============');
console.log('-- getParams 获取所有地址参数 --');
const url_01 = "https://www.baidu.com?name=zs&age=20&hobbies[]=reading&hobbies[]=sports";
const params = Lib.url.getParams(url_01);
console.log(params); // { name: 'zs', age: '20', hobbies: [ 'reading', 'sports' ] }

console.log('-- getParam 获取指定地址参数 --');
const name = Lib.url.getParam(url_01, 'name');
console.log(name); // zs

console.log('-- appendParam 地址添加参数 --');
const appendUrl = Lib.url.appendParam(url_01, "test", "value");
console.log(appendUrl); // https://www.baidu.com?name=zs&age=20&hobbies[]=reading&hobbies[]=sports&test=value

console.log('-- transParams 对象转换参数 --');
const transParams = {
  name: 'zs',
  age: 20,
  sex: 0
}
const transUrl = Lib.url.transParams(transParams);
console.log(transUrl); //name=zs&age=20&sex=0

stringKit 字符串处理 ​

  • 字符串脱敏
  • 字符串去空格
  • 获取随机手机号
  • 获取随机uuid
js
import {stringKit} from "y-js-kit";

console.log('============ stringKit ============');
console.log('-- desensitization 脱敏 --');
console.log(stringKit.desensitization("123456789", 2, 4)); // 12****789

console.log('-- trim 去空格 --');
console.log(stringKit.trim(" 123 "));// 123
console.log(stringKit.trim("12 345", "all")); // 12345
console.log(stringKit.trim(" 123124","left")); // 123124

console.log('-- getMobile 随机获取手机号 --');
console.log(stringKit.getMobile()); // 13095692650

console.log('-- getUuid 随机uuid --');
console.log(stringKit.getUuid()); //89ff8a90-98b6-469a-b511-8bc10db746da

objKit 对象处理 ​

  • 对象合并
  • 过滤属性
  • 过滤空值
js
import {objKit} from "y-js-kit";

console.log('============ objKit ============');
console.log('-- objMerge 对象合并 --');
const target = {
  a: 1,
  b: 2,
  c: 3,
}
const source = {
  c: 4,
  d: 5,
  e: null
}
const obj = objKit.objMerge(target, source);
const obj02 = objKit.objMerge(target, source, 'soft');
console.log(obj); // { a: 1, b: 2, c: 4, d: 5 }
console.log(obj02); //{ a: 1, b: 2, c: 4 }

console.log('-- filterProperties 属性过滤 --');
console.log(objKit.filterProperties(obj, ['a'])); //{ b: 2, c: 4, d: 5 }

console.log('-- filterNull 过滤空值 --');
console.log(objKit.filterNull(source)); // { c: 4, d: 5 }

commonKit 通用工具 ​

  • 睡眠
  • 防抖
  • 节流
  • 类型判断
  • 深拷贝
js
import {commonKit} from "y-js-kit";

console.log('============ commonKit ============');
console.log('-- sleep 睡眠 --');
await commonKit.sleep(1000);

console.log('-- deepClone --');
const obj1 = {a: 1, b: 2};
const obj2 = commonKit.deepClone(obj1);
console.log(obj1 === obj2);


console.log('-- getType --');
console.log(commonKit.getType(1)); //number
console.log(commonKit.getType("1")); //string
console.log(commonKit.getType(true)); //boolean
console.log(commonKit.getType(() => {
})); // function
console.log(commonKit.getType({name: "zs"})); // object

console.log('-- throttle --');
commonKit.throttle(() => {
  console.log("233")
})

console.log('-- debounce --');
commonKit.debounce(() => {
  console.log("233")
})

arrKit 数组处理 ​

  • 数组填充
  • 数组去重
  • 数组分页
  • 删除元素
js
import { arrKit } from 'y-js-kit';

console.log('============ arrKit ============');
console.log('-- arrayFill 数组填充 --');
const arrayFill = arrKit.arrayFill([1, 2, 3], 10, null);
console.log(JSON.stringify(arrayFill)); // [1,2,3,null,null,null,null,null,null,null]

console.log('-- arrayUnique 去重 --');
const arrayUnique = arrKit.arrayUnique([1, 2, 3, 4, 5, 5, 5, 55, 5, 5]);
console.log(arrayUnique); //[ 1, 2, 3, 4, 5, 55 ]

console.log('-- paginate 分页 --');
const paginate = arrKit.paginate(arrayFill, 1, 3);
console.log(paginate); // [ 1, 2, 3 ]

console.log('-- delElement 删除 --');
const delElement = arrKit.delElement([1, 2, 3, 4, 5, 6], 2);
console.log(delElement); // [ 1, 3, 4, 5, 6 ]
const arr = [
  {
    id: 1,
    name: 'zs',
  },
  {
    id: 2,
    name: 'xs',
  },
  {
    id: 3,
    name: 'ls',
  },
];
// 按照相同的属性进行删除
const delElement1 = arrKit.delElement(arr, { id: 1, name: 'zs' });
console.log(delElement1); // [ { id: 2, name: 'xs' }, { id: 3, name: 'ls' } ]
// 按指定key的值进行匹配删除
const delElement2 = arrKit.delElement(arr, { name: 'ls' }, 'name');
console.log(delElement2); // [ { id: 1, name: 'zs' }, { id: 2, name: 'xs' } ]
